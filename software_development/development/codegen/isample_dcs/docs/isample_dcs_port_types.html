isample_dcs_port_types.h¶

Due to the nature of C++ the data types to be interchanged using msgpack need to be declared and defined at compile time. This file includes the list of specific C++ data types required for this DCS data ports.

Code: isample_dcs_port_types.h¶
#ifndef _isample_dcs_port_types_h_
#define _isample_dcs_port_types_h_

#include "HeartBeatEvent.h"
#include "isample_temp_measurements.h"
#include "isample_hmi_buttons.h"
#include "isample_motor_status.h"
#include "isample_hmi_leds.h"
#include "isample_motor_control.h"
#include "isample_sdo_data.h"
#include "FieldBus.h"

#endif // _isample_dcs_port_types_h_
